Windows 右键bush here 失效

  • 运行regedit.exe进入注册表，在HKEY_CLASSES_ROOT\Directory\Background\shell中进行设置。
  • 新建项Git Bush Here，此时你可以看到在桌面右键会出现“Git Bush Here”菜单。
  • 添加Git Bush Icon，在第一步的新建项Git Bush Here下，新建字符串值Icon，然后编辑该值为“C:\Program Files\Git\mingw64\share\git\git-for-windows.ico”，你需要根据你安装的Git 目录进行配置修改。完成此步后，你会发现右键菜单“Git Bush Here”会多出一个Icon。
  • 添加Command项。在“Git Bush Here”下再新建项“Command”，将其默认值改为“C:\Program Files\Git\bin\bash.exe --login -i”，这样，你就可以通过右键菜单的方式快速进入Git命令行工具，进行代码版本管理。

scp命令从远程下载文件 (秘钥方式)

// 下载单个文件
scp -i ~/ssh.key root@IP:data/ ./
// 下载目录
scp -r -i ~/ssh.key root@IP:data/ ./

// 上传时本地目录和远程目录顺序调换即可

树形<=>扁平

/**
 * @function 扁平化结构数据转换为树形结构数据
 * @param {*} obj 源数据
 * @param {*} childrenKey 需要遍历的子数据的key
 * @param {*} parentId 用于比较的父级id的key
 * @param {*} childId 子级中用于和父级比较的key
 * @returns 树结构
 */
export function formatTree(obj, childrenKey = 'children', parentId = 'gid', childId = 'parentGid') {
	let copyedObj = JSON.parse(JSON.stringify(obj)) //深拷贝源数据
	return copyedObj.filter(parent => {
		parent.id = parent[parentId]
		let findChildren = copyedObj.filter(child => {
			return parent[parentId] == child[childId]
		})
		findChildren.length > 0 ? parent[childrenKey] = findChildren : parent[childrenKey] = []
		return !parent[childId] //返回顶层，依据实际情况判断这里的返回值
	})
}

/**
 * @function 树形结构转为扁平结构
 * @param {*} data 源数据
 * @param {*} childrenKey 需要遍历的子数据的key
 * @returns 树结构
 */
export function formatList(data, childrenKey = 'childs') {
	return data.reduce((res, node) => {
		res.push(node)
		if (node[childrenKey]) res = res.concat(formatList(node[childrenKey], childrenKey));
		return res
	}, [])
}
